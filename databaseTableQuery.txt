//서울 크롤링 데이터 저장
CREATE TABLE `seoul` (
   `id` BIGINT NOT NULL AUTO_INCREMENT,
   `department1Id` BIGINT NULL DEFAULT NULL,
   `department2Id` BIGINT NULL DEFAULT NULL,
   `department3Id` BIGINT NULL DEFAULT NULL,
   `department4Id` BIGINT NULL DEFAULT NULL,
   `department5Id` BIGINT NULL DEFAULT NULL,
   `department` VARCHAR(50) NULL DEFAULT NULL,
   `usedAt` DATETIME NULL DEFAULT NULL,
   `place` VARCHAR(200) NULL DEFAULT NULL,
   `usePurpose` VARCHAR(200) NULL DEFAULT NULL,
   `price` INT NULL DEFAULT NULL,
   `user` VARCHAR(100) NULL DEFAULT NULL,
   `paymentMethod` VARCHAR(100) NULL DEFAULT NULL,
   `crawlDate` DATETIME NULL DEFAULT NULL,
   PRIMARY KEY (`id`)
)
COLLATE='utf8mb4_0900_ai_ci';
